import { Playground, Props } from 'docz';
import { Breadcrumb, Crumb } from '@westpac/breadcrumb';
import { useHistory } from 'react-router-dom';

# Breadcrumbs

Breadcrumbs are styled navigational links used to indicate a user journey or path.
They are a simple, effective and proven method to aid orientation.

<Playground>
	<Breadcrumb current="/credit-cards/">
		<Crumb href="/" text="Home" />
		<Crumb href="/personal-banking/" text="Personal" />
		<Crumb href="/credit-cards/" text="Credit cards" />
	</Breadcrumb>
</Playground>

You can use the component with your favorite router as well

<Playground>
	{() => {
		const history = useHistory(); 
		
		return (
			<Breadcrumb current="/credit-cards/">
				<Crumb text="Home" onClick={() => history.push('/')} />
				<Crumb text="Personal" onClick={() => history.push('/personal-banking/')} />
				<Crumb text="Credit cards" onClick={() => history.push('/credit-cards/')} />
			</Breadcrumb>
		);
	}}
</Playground>

Or drive your breadcrumbs from data directly

<Playground>
	{() => {
		const history = useHistory();

		return (
			<Breadcrumb
				current="/credit-cards/"
				data={[
					{ href: '/', text: 'Home' },
					{ href: '/personal-banking/', text: 'Personal' },
					{ text: 'Credit cards', onClick: () => history.push('/credit-cards/') },
				]}
			/>
		);
	}}
</Playground>
